10. Створіть масив, наповніть його даниими в хаотичному порядку, знайдіть найбільше та найменше значення масиву 
public class Masyv {

	public static void main(String[] args) {
		
		int[] chysla = {7, 9, 15, 13, 18};
		
		int maxVal = chysla[0];
		int minVal = chysla[0];
		for (int i = 0; i < chysla.length; i++ ) {
			
			if (maxVal < chysla[i]) {
				maxVal = chysla[i];
			}
			if (minVal > chysla[i]) {
				minVal = chysla[i];
			}
			
		}
		System.out.println("Max: "+maxVal);
		System.out.println("Min: "+minVal);
	}

}
